**Goal:** Make Tennis Tagger actually functional — real Python ML inference via subprocess bridge, concurrent pipeline for performance, model downloading, and a Wails+Svelte desktop UI.

**Architecture:** ProcessBridge communicates with a Python subprocess via stdin/stdout JSON-RPC. The pipeline becomes concurrent with 3-stage goroutine channels. Wails wraps the Go backend with a Svelte frontend. Models download on first run.

**Tech Stack:** Go 1.22+, Wails v2, Svelte 4, Python 3.11, ffmpeg, PyTorch 2.4, Ultralytics 8.3

**Depends on:** Phase 1 complete (all Go packages + Python ML modules exist)

## Scope Decision: ProcessBridge First, Not EmbeddedBridge

The spec allows both embedded (go-python3/cgo) and process (subprocess) bridges. We implement **ProcessBridge first** because:
- No cgo complexity (avoids gcc/MinGW on Windows, CUDA thread issues)
- Easier to debug (two separate processes)
- Python side can use any library without embedding constraints
- Same BridgeBackend interface — can swap to embedded later as optimization

Transport: **stdin/stdout JSON-RPC** (simplest cross-platform IPC, no named pipes needed for Phase 1.5).

**Known performance gap:** Base64 frame encoding inflates data ~33%. A 32-frame batch at 1080p is ~256MB through a text pipe. This is acceptable for Phase 1.5 correctness; a follow-up task should implement named-pipe + mmap transport for production performance before the Phase 1 exit criterion ("faster than current Python app on 60-minute video").

```python
#!/usr/bin/env python3
"""
Tennis Tagger ML Bridge Server.
Reads JSON-RPC requests from stdin, writes responses to stdout.
Logs go to stderr (captured by Go as diagnostics).
"""
import json
import sys
import logging
import traceback
import numpy as np

# Redirect all logging to stderr so stdout stays clean for JSON-RPC
logging.basicConfig(stream=sys.stderr, format="%(levelname)s %(name)s %(message)s", level=logging.INFO)
logger = logging.getLogger("bridge")


class BridgeServer:
    """Handles JSON-RPC requests by dispatching to ML modules."""

    def __init__(self):
        self.detector = None
        self.pose_estimator = None
        self.classifier = None
        self.analyzer = None
        self.score_tracker = None
        self.trainer = None
        self._initialized = False

    def handle_init(self, params: dict) -> dict:
        """Initialize all ML modules."""
        import torch
        torch.set_num_threads(1)  # Spec requirement: avoid ATen/Go thread conflict

        models_dir = params.get("ModelsDir", "models")
        device = params.get("Device", "auto")

        logger.info("Initializing ML modules (device=%s, models=%s)", device, models_dir)

        from ml.detector import Detector
        from ml.pose import PoseEstimator
        from ml.classifier import StrokeClassifier
        from ml.analyzer import Analyzer
        from ml.score import ScoreTracker
        from ml.trainer import Trainer

        import os
        detector_path = os.path.join(models_dir, "yolov8x.pt")
        stroke_path = os.path.join(models_dir, "stroke_x3d.pt")

        self.detector = Detector(detector_path, device)
        self.pose_estimator = PoseEstimator(device)
        self.classifier = StrokeClassifier(stroke_path, device)
        self.analyzer = Analyzer()
        self.score_tracker = ScoreTracker(device)
        self.trainer = Trainer(models_dir, device)
        self._initialized = True

        logger.info("All ML modules initialized")
        return {"status": "ok"}

    def handle_detect_batch(self, params: dict) -> list:
        """Detect players and ball in frames.

        params: {"frames": [{"width": W, "height": H, "data_base64": "..."}], ...}
        For performance, frames are passed as base64-encoded RGB bytes.
        """
        import base64
        frames_data = params.get("frames", [])
        frames = []
        for fd in frames_data:
            raw = base64.b64decode(fd["data_base64"])
            arr = np.frombuffer(raw, dtype=np.uint8).reshape(fd["height"], fd["width"], 3)
            frames.append(arr)

        if not frames:
            return []

        batch = np.stack(frames)
        return self.detector.detect_batch(batch)

    def handle_detect_court(self, params: dict) -> dict:
        """Detect court from a single frame."""
        import base64
        fd = params["frame"]
        raw = base64.b64decode(fd["data_base64"])
        frame = np.frombuffer(raw, dtype=np.uint8).reshape(fd["height"], fd["width"], 3)
        return self.analyzer.detect_court(frame)

    def handle_classify_strokes(self, params: dict) -> list:
        """Classify stroke types from frame clips."""
        import base64
        clips_data = params.get("clips", [])
        if not clips_data:
            return []

        clips = []
        for cd in clips_data:
            frames = []
            for fd in cd["frames"]:
                raw = base64.b64decode(fd["data_base64"])
                arr = np.frombuffer(raw, dtype=np.uint8).reshape(fd["height"], fd["width"], 3)
                frames.append(arr)
            clips.append(np.stack(frames))

        batch = np.stack(clips)
        return self.classifier.classify(batch)

    def handle_analyze_placements(self, params: dict) -> list:
        """Analyze shot placements."""
        detections = params.get("detections", [])
        court = params.get("court", {})
        return self.analyzer.analyze_placements(detections, court)

    def handle_segment_rallies(self, params: dict) -> list:
        """Segment detections into rallies."""
        detections = params.get("detections", [])
        fps = params.get("fps", 30.0)
        return self.analyzer.segment_rallies(detections, fps)

    def dispatch(self, method: str, params: dict):
        """Route a method call to the appropriate handler."""
        handlers = {
            "__init__": self.handle_init,
            "detect_batch": self.handle_detect_batch,
            "detect_court": self.handle_detect_court,
            "classify_strokes": self.handle_classify_strokes,
            "analyze_placements": self.handle_analyze_placements,
            "segment_rallies": self.handle_segment_rallies,
        }
        handler = handlers.get(method)
        if handler is None:
            raise ValueError(f"Unknown method: {method}")
        return handler(params)


def main():
    """Main loop: read JSON-RPC from stdin, write responses to stdout."""
    import torch  # For CUDA OOM handling in dispatch loop
    server = BridgeServer()
    logger.info("Bridge server started, waiting for requests...")

    # Signal readiness
    sys.stdout.write(json.dumps({"ready": True}) + "\n")
    sys.stdout.flush()

    for line in sys.stdin:
        line = line.strip()
        if not line:
            continue

        try:
            request = json.loads(line)
        except json.JSONDecodeError as e:
            sys.stdout.write(json.dumps({"error": f"Invalid JSON: {e}", "id": None}) + "\n")
            sys.stdout.flush()
            continue

        req_id = request.get("id")
        method = request.get("method", "")
        params = request.get("params", {})

        try:
            result = server.dispatch(method, params)
            response = {"result": result, "id": req_id}
        except torch.cuda.OutOfMemoryError as e:
            # Signal CUDA OOM specifically so Go can retry with smaller batch
            logger.error("CUDA OOM: %s", e)
            torch.cuda.empty_cache()
            response = {"error": f"CUDA_OOM: {e}", "id": req_id}
        except Exception as e:
            logger.error("Error handling %s: %s", method, e)
            traceback.print_exc(file=sys.stderr)
            response = {"error": str(e), "id": req_id}

        sys.stdout.write(json.dumps(response, default=_json_default) + "\n")
        sys.stdout.flush()


def _json_default(obj):
    """Handle numpy types in JSON serialization."""
    if isinstance(obj, np.integer):
        return int(obj)
    if isinstance(obj, np.floating):
        return float(obj)
    if isinstance(obj, np.ndarray):
        return obj.tolist()
    raise TypeError(f"Object of type {type(obj)} is not JSON serializable")


if __name__ == "__main__":
    main()
```

```go
// ProcessBridge implements BridgeBackend via a Python subprocess.
type ProcessBridge struct {
    worker *Worker
}

func NewProcessBridge(pythonPath string) *ProcessBridge {
    caller := &processCaller{pythonPath: pythonPath}
    return &ProcessBridge{
        worker: NewWorker(caller),
    }
}

func (b *ProcessBridge) Init(config BridgeConfig) error {
    payload, _ := json.Marshal(config)
    _, err := b.worker.Call("__init__", payload)
    return err
}

func (b *ProcessBridge) DetectBatch(frames []Frame) ([]DetectionResult, error) {
    // Encode frames as base64 for JSON transport
    frameData := make([]map[string]interface{}, len(frames))
    for i, f := range frames {
        frameData[i] = map[string]interface{}{
            "width":       f.Width,
            "height":      f.Height,
            "data_base64": base64.StdEncoding.EncodeToString(f.Data),
        }
    }
    payload, _ := json.Marshal(map[string]interface{}{"frames": frameData})

    resp, err := b.worker.Call("detect_batch", payload)
    if err != nil {
        return nil, err
    }

    var results []DetectionResult
    if err := json.Unmarshal(resp, &results); err != nil {
        // The Python detector returns a different shape — adapt
        var rawResults []map[string]interface{}
        if err2 := json.Unmarshal(resp, &rawResults); err2 != nil {
            return nil, fmt.Errorf("unmarshal detect results: %w (raw: %s)", err, string(resp))
        }
        results = adaptDetectionResults(rawResults)
    }
    return results, nil
}

func (b *ProcessBridge) ClassifyStrokes(clips []FrameClip) ([]StrokeResult, error) {

// adaptDetectionResults converts Python detector output format to Go DetectionResult.
func adaptDetectionResults(raw []map[string]interface{}) []DetectionResult {
    results := make([]DetectionResult, len(raw))
    for i, r := range raw {
        det := DetectionResult{FrameIndex: i}

        // Parse players
        if players, ok := r["players"].([]interface{}); ok {
            for _, p := range players {
                pm, _ := p.(map[string]interface{})
                bbox, _ := pm["bbox"].([]interface{})
                if len(bbox) == 4 {
                    det.Players = append(det.Players, BBox{
                        X1: toFloat(bbox[0]), Y1: toFloat(bbox[1]),
                        X2: toFloat(bbox[2]), Y2: toFloat(bbox[3]),
                        Confidence: toFloat(pm["confidence"]),
                    })
                }
            }
        }

        // Parse ball
        if ball, ok := r["ball"].(map[string]interface{}); ok && ball != nil {
            bbox, _ := ball["bbox"].([]interface{})
            if len(bbox) == 4 {
                b := BBox{
                    X1: toFloat(bbox[0]), Y1: toFloat(bbox[1]),
                    X2: toFloat(bbox[2]), Y2: toFloat(bbox[3]),
                    Confidence: toFloat(ball["confidence"]),
                }
                det.Ball = &b
            }
        }

        results[i] = det
    }
    return results
}

func toFloat(v interface{}) float64 {
    switch n := v.(type) {
    case float64:
        return n
    case int:
        return float64(n)
    case json.Number:
        f, _ := n.Float64()
        return f
    }
    return 0
}

// --- processCaller: manages the Python subprocess ---

type processCaller struct {
    pythonPath string
    cmd        *exec.Cmd
    stdin      io.WriteCloser
    scanner    *bufio.Scanner
    mu         sync.Mutex
    nextID     atomic.Int64
}

func (c *processCaller) init(config BridgeConfig) error {
    // init is called by Worker — just forward as JSON-RPC __init__ to Python
    if c.stdin == nil {
        if err := c.startProcess(); err != nil {
            return err
        }
    }
    payload, _ := json.Marshal(config)
    _, err := c.callRaw("__init__", payload)
    return err
}

func (c *processCaller) startProcess() error {
    // Start Python bridge server
    c.cmd = exec.Command(c.pythonPath, "-m", "ml.bridge_server")
    c.cmd.Dir = "." // Working directory should be project root

    var err error
    c.stdin, err = c.cmd.StdinPipe()
    if err != nil {
        return fmt.Errorf("stdin pipe: %w", err)
    }

    stdout, err := c.cmd.StdoutPipe()
    if err != nil {
        return fmt.Errorf("stdout pipe: %w", err)
    }

    // stderr goes to Go's stderr for diagnostics
    c.cmd.Stderr = nil // inherits parent stderr

    if err := c.cmd.Start(); err != nil {
        return fmt.Errorf("start python: %w", err)
    }

    c.scanner = bufio.NewScanner(stdout)
    // 512MB buffer — a 32-frame batch at 1080p base64 is ~256MB per response line
    c.scanner.Buffer(make([]byte, 64*1024), 512*1024*1024)

    // Wait for ready signal
    if !c.scanner.Scan() {
        return fmt.Errorf("python process did not send ready signal")
    }
    var ready map[string]interface{}
    if err := json.Unmarshal(c.scanner.Bytes(), &ready); err != nil {
        return fmt.Errorf("invalid ready signal: %w", err)
    }
    if ready["ready"] != true {
        return fmt.Errorf("unexpected ready signal: %v", ready)
    }

    return nil
}

func (c *processCaller) call(method string, payload json.RawMessage) (json.RawMessage, error) {
    c.mu.Lock()
    defer c.mu.Unlock()

    // Start subprocess on first call (lazy init)
    if c.stdin == nil {
        if err := c.startProcess(); err != nil {
            return nil, fmt.Errorf("start python process: %w", err)
        }
    }

    id := c.nextID.Add(1)

    request := map[string]interface{}{
        "method": method,
        "params": json.RawMessage(payload),
        "id":     id,
    }

    reqBytes, err := json.Marshal(request)
    if err != nil {
        return nil, fmt.Errorf("marshal request: %w", err)
    }

    return c.sendAndReceive(reqBytes)
}

// sendAndReceive writes a JSON-RPC request and reads the response.
func (c *processCaller) sendAndReceive(reqBytes []byte) (json.RawMessage, error) {
    if _, err := c.stdin.Write(append(reqBytes, '\n')); err != nil {
        return nil, fmt.Errorf("write to python: %w", err)
    }

    if !c.scanner.Scan() {
        return nil, fmt.Errorf("python process closed stdout")
    }

    var response struct {
        Result json.RawMessage `json:"result"`
        Error  *string         `json:"error"`
        ID     int64           `json:"id"`
    }
    if err := json.Unmarshal(c.scanner.Bytes(), &response); err != nil {
        return nil, fmt.Errorf("unmarshal response: %w (raw: %s)", err, c.scanner.Text())
    }

    if response.Error != nil {
        errMsg := *response.Error
        // Detect CUDA OOM for retry logic in pipeline
        if len(errMsg) > 9 && errMsg[:9] == "CUDA_OOM:" {
            return nil, fmt.Errorf("CUDA_OOM: %s", errMsg[10:])
        }
        return nil, fmt.Errorf("python error: %s", errMsg)
    }

    return response.Result, nil
}

// callRaw sends a JSON-RPC call without lazy process start (used by init).
func (c *processCaller) callRaw(method string, payload json.RawMessage) (json.RawMessage, error) {
    id := c.nextID.Add(1)
    request := map[string]interface{}{
        "method": method,
        "params": json.RawMessage(payload),
        "id":     id,
    }
    reqBytes, _ := json.Marshal(request)
    return c.sendAndReceive(reqBytes)
}

func (c *processCaller) close() {
    if c.stdin != nil {
        c.stdin.Close()
    }
    if c.cmd != nil && c.cmd.Process != nil {
        c.cmd.Process.Kill()
        c.cmd.Wait()
    }
}

var _ BridgeBackend = (*ProcessBridge)(nil)
```

```go
package main

import (
    "flag"
    "fmt"
    "log/slog"
    "os"

    "github.com/liamp/tennis-tagger/internal/app"
    "github.com/liamp/tennis-tagger/internal/bridge"
    "github.com/liamp/tennis-tagger/internal/config"
)

func main() {
    slog.SetDefault(slog.New(slog.NewJSONHandler(os.Stderr, &slog.HandlerOptions{Level: slog.LevelInfo})))

    useMock := flag.Bool("mock", false, "Use mock bridge instead of real Python ML")
    pythonPath := flag.String("python", "python", "Path to Python executable")
    flag.Parse()

    cfg, err := config.Load("")
    if err != nil {
        slog.Error("Failed to load config", "error", err)
        os.Exit(1)
    }

    var b bridge.BridgeBackend
    if *useMock {
        b = bridge.NewMockBridge()
        fmt.Println("Using mock bridge (no Python)")
    } else {
        b = bridge.NewProcessBridge(*pythonPath)
        fmt.Println("Using Python process bridge")
    }

    a := app.NewApp(cfg, b)

    args := flag.Args()
    if len(args) > 0 {
        videoPath := args[0]
        fmt.Printf("Processing: %s\n", videoPath)

        if err := b.Init(bridge.BridgeConfig{
            ModelsDir: cfg.ModelsDir,
            Device:    cfg.Device,
        }); err != nil {
            slog.Error("Bridge init failed", "error", err)
            fmt.Println("Falling back to mock bridge")
            b = bridge.NewMockBridge()
            a = app.NewApp(cfg, b)
        }

        if err := a.ProcessVideo(videoPath); err != nil {
            slog.Error("Processing failed", "error", err)
            os.Exit(1)
        }

        progress := a.GetProgress()
        fmt.Printf("Done: %d/%d frames\n", progress.ProcessedFrames, progress.TotalFrames)

        outputPath := videoPath + "_output.csv"
        if err := a.ExportCSV(outputPath); err != nil {
            slog.Error("Export failed", "error", err)
            os.Exit(1)
        }
        fmt.Printf("CSV: %s\n", outputPath)
    } else {
        fmt.Println("Tennis Tagger v2")
        fmt.Println("Usage: tagger [--mock] [--python path] <video.mp4>")
    }

    b.Close()
}
```

```go
// internal/models/download.go
package models

import (
    "crypto/sha256"
    "encoding/hex"
    "fmt"
    "io"
    "net/http"
    "os"
    "path/filepath"
)

// ProgressFunc is called during download with bytes downloaded and total size.
type ProgressFunc func(downloaded, total int64)

// DownloadModel downloads a single model to destDir with checksum verification.
func DownloadModel(model Model, baseURL, destDir string, progress ProgressFunc) error {
    destPath := filepath.Join(destDir, model.File)

    // Skip if already exists and checksum matches
    if model.SHA256 != "" {
        if existing, err := checksumFile(destPath); err == nil && existing == model.SHA256 {
            return nil
        }
    }

    url := baseURL + model.File
    resp, err := http.Get(url)
    if err != nil {
        return fmt.Errorf("download %s: %w", model.File, err)
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusOK {
        return fmt.Errorf("download %s: HTTP %d", model.File, resp.StatusCode)
    }

    // Write to temp file
    tmpPath := destPath + ".tmp"
    f, err := os.Create(tmpPath)
    if err != nil {
        return fmt.Errorf("create temp file: %w", err)
    }

    hasher := sha256.New()
    writer := io.MultiWriter(f, hasher)

    var downloaded int64
    buf := make([]byte, 32*1024)
    for {
        n, readErr := resp.Body.Read(buf)
        if n > 0 {
            if _, writeErr := writer.Write(buf[:n]); writeErr != nil {
                f.Close()
                os.Remove(tmpPath)
                return fmt.Errorf("write: %w", writeErr)
            }
            downloaded += int64(n)
            if progress != nil {
                progress(downloaded, model.Size)
            }
        }
        if readErr == io.EOF {
            break
        }
        if readErr != nil {
            f.Close()
            os.Remove(tmpPath)
            return fmt.Errorf("read: %w", readErr)
        }
    }
    f.Close()

    // Verify checksum
    if model.SHA256 != "" {
        got := hex.EncodeToString(hasher.Sum(nil))
        if got != model.SHA256 {
            os.Remove(tmpPath)
            return fmt.Errorf("checksum mismatch for %s: expected %s, got %s", model.File, model.SHA256, got)
        }
    }

    // Atomic rename
    if err := os.Rename(tmpPath, destPath); err != nil {
        return fmt.Errorf("rename: %w", err)
    }

    return nil
}

// DownloadMissing downloads all missing required models.
func DownloadMissing(manifest *Manifest, baseURL, modelsDir string, progress ProgressFunc) error {
    missing := CheckModels(manifest, modelsDir)
    if len(missing) == 0 {
        return nil
    }

    if err := os.MkdirAll(modelsDir, 0755); err != nil {
        return err
    }

    for _, model := range missing {
        if err := DownloadModel(model, baseURL, modelsDir, progress); err != nil {
            return err
        }
    }
    return nil
}

func checksumFile(path string) (string, error) {
    f, err := os.Open(path)
    if err != nil {
        return "", err
    }
    defer f.Close()
    h := sha256.New()
    io.Copy(h, f)
    return hex.EncodeToString(h.Sum(nil)), nil
}
```

```go
// internal/pipeline/concurrent.go
package pipeline

import (
    "fmt"
    "log/slog"
    "sync"

    "github.com/liamp/tennis-tagger/internal/bridge"
    "github.com/liamp/tennis-tagger/internal/config"
    "github.com/liamp/tennis-tagger/internal/tracker"
    "github.com/liamp/tennis-tagger/internal/video"
)

// frameBatch is passed from Stage 1 to Stage 2.
type frameBatch struct {
    Frames     []video.Frame
    StartFrame int
}

// detectionBatch is passed from Stage 2 to Stage 3.
type detectionBatch struct {
    Detections []bridge.DetectionResult
    StartFrame int
}

// ProcessConcurrent runs the 3-stage pipeline with goroutine channels.
func (p *Pipeline) ProcessConcurrent(videoPath string) (*Result, error) {
    slog.Info("Starting concurrent pipeline", "video", videoPath)

    vr, err := video.Open(videoPath)
    if err != nil {
        return nil, fmt.Errorf("open video: %w", err)
    }
    defer vr.Close()

    meta := vr.Metadata()
    result := &Result{
        VideoPath:   videoPath,
        TotalFrames: meta.TotalFrames,
        FPS:         meta.FPS,
    }

    p.progress = ProgressInfo{TotalFrames: meta.TotalFrames, Stage: "initializing"}
    p.tracker.Reset()

    batchSize := p.config.Pipeline.BatchSize
    if batchSize <= 0 {
        batchSize = 32
    }

    // Court detection (once, from first frame)
    firstFrames, err := vr.ExtractBatch(0, 1)
    if err == nil && len(firstFrames) > 0 {
        court, err := p.bridge.DetectCourt(bridge.Frame{
            Data: firstFrames[0].Data, Width: firstFrames[0].Width, Height: firstFrames[0].Height,
        })
        if err == nil {
            result.Court = court
        }
    }

    // Channels with backpressure (buffer 2 batches)
    frameCh := make(chan frameBatch, 2)
    detCh := make(chan detectionBatch, 2)
    var wg sync.WaitGroup

    // Stage 1: Frame extraction (goroutine)
    wg.Add(1)
    go func() {
        defer wg.Done()
        defer close(frameCh)
        p.progress.Stage = "extracting"

        for start := 0; start < meta.TotalFrames; start += batchSize {
            count := batchSize
            if start+count > meta.TotalFrames {
                count = meta.TotalFrames - start
            }

            frames, err := vr.ExtractBatch(start, count)
            if err != nil || len(frames) == 0 {
                slog.Warn("Frame extraction stopped", "frame", start, "error", err)
                return
            }

            frameCh <- frameBatch{Frames: frames, StartFrame: start}
        }
    }()

    // Stage 2: ML inference (goroutine — bridge worker handles thread pinning)
    wg.Add(1)
    go func() {
        defer wg.Done()
        defer close(detCh)
        p.progress.Stage = "detecting"

        for batch := range frameCh {
            bFrames := make([]bridge.Frame, len(batch.Frames))
            for i, f := range batch.Frames {
                bFrames[i] = bridge.Frame{Data: f.Data, Width: f.Width, Height: f.Height}
            }

            detections, err := p.bridge.DetectBatch(bFrames)
            if err != nil {
                slog.Error("Detection failed", "error", err)
                continue
            }

            for i := range detections {
                detections[i].FrameIndex = batch.StartFrame + i
                detections[i].Players = filterBBoxes(detections[i].Players)
            }

            detCh <- detectionBatch{Detections: detections, StartFrame: batch.StartFrame}
        }
    }()

    // Stage 3: Tracking + assembly (main goroutine — sole writer to result, no mutex needed)
    p.progress.Stage = "tracking"
    for batch := range detCh {
        batchTracks := make([][]tracker.TrackedObject, len(batch.Detections))
        for i, det := range batch.Detections {
            batchTracks[i] = p.tracker.Update(det.Players)
        }

        result.Detections = append(result.Detections, batch.Detections...)
        result.Tracks = append(result.Tracks, batchTracks...)
        result.ProcessedFrames += len(batch.Detections)
        // Progress reads from UI are best-effort/racy — acceptable for progress display
        p.progress.ProcessedFrames = result.ProcessedFrames
        p.progress.Percent = float64(result.ProcessedFrames) / float64(meta.TotalFrames) * 100

        // Checkpoint
        if p.config.Pipeline.CheckpointEvery > 0 && result.ProcessedFrames%p.config.Pipeline.CheckpointEvery == 0 {
            cpPath := checkpointPath(videoPath)
            SaveCheckpoint(&Checkpoint{
                VideoPath:       videoPath,
                ProcessedFrames: result.ProcessedFrames,
                TotalFrames:     meta.TotalFrames,
            }, cpPath)
        }
    }

    wg.Wait()

    // Post-processing
    if len(result.Detections) > 0 {
        if rallies, err := p.bridge.SegmentRallies(result.Detections, meta.FPS); err == nil {
            result.Rallies = rallies
        }
        if placements, err := p.bridge.AnalyzePlacements(result.Detections, result.Court); err == nil {
            result.Placements = placements
        }
    }

    p.progress.Stage = "complete"
    p.progress.Percent = 100
    slog.Info("Concurrent pipeline complete", "processed", result.ProcessedFrames)
    return result, nil
}
```

```go
    go func() {
        if err := a.ProcessVideo(path); err != nil {
            wailsRuntime.EventsEmit(a.ctx, "processing-error", err.Error())
            return
        }
        wailsRuntime.EventsEmit(a.ctx, "processing-complete", nil)
    }()
}
```
